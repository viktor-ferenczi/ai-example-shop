```cshtml
<table class="table table-hover table-bordered" id="categoryIndexTable">
    <caption>List of food items in the @Model.Category.Name category</caption>
    <thead>
        <tr>
            <th>Food</th>
        </tr>
    </thead>
    <tbody>
        @foreach (var food in Model.Foods)
        {
            <tr>
                <td>
                   @Html.Partial("FoodSummary",food) 
                </td>
            </tr>
        }
    </tbody>
</table>
```
