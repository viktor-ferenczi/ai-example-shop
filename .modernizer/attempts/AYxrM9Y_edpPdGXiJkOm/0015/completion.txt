 
The issue with the `GetFilteredFoods` method parameter naming is that it's not clear what the `q` parameter represents. It's not immediately obvious what it's searching for.

public IEnumerable<Food> GetFilteredFoods(string searchQuery)
{
    const int defaultCount = 10;
    var queries = string.IsNullOrEmpty(searchQuery) ? null : Regex.Replace(searchQuery, @"\s+", " ").Trim().ToLower().Split(" ");
    if (queries == null)
    {
        return GetPreferred(defaultCount);
    }

    return GetAll().Where(item => queries.Any(query => (item.Name.ToLower().Contains(query))));
}
